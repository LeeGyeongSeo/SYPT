getchar

<설명> 
표준 입력 장치로부터 문자 하나를 입력 받는다.
문자를 입력한 후에 엔터키를 만나거나 파일의 끝과 같은 Ctrl-D를 만났을 때 문자값을 하나씩 받아 온다.
헤더 stdio.h 
형태 int getchar(void); 
 

<인수>  
int  입력받은 문자, 파일의 끝이면 EOF를 반환  
 

<예제> 
#include <stdio.h>
int main( void)
{
   char   ch_key;

   while( 1)
   {       
      ch_key   = getchar();          // 문자 하나씩 받아서
      if ( EOF == ch_key)            // 입력의 끝이면
      {
         printf( "nquitn");        // 입력 종료를 알리고
         break;                     // 루프 종료 
      }
      else
         printf( "-%c", ch_key);     // 입력 받은 문자 앞에 -를 추가
   }
   return 0;
}
